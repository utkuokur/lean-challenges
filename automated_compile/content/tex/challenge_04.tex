\section*{Challenge 4: Sidorenko Conjecture for Bipartite Graphs with Bounded Sides}
\refstepcounter{section}

% \cite{ }

\begin{definition}[Graph homomorphism density]
Let $G$ and $H$ be simple graphs. 
%\phantom{a}
\begin{enumerate}
\item A function $f: V\ev{H} \rightarrow V\ev{G}$ is a \textit{graph homomorphism} from $H$ to $G$ if for each edge $uv \in E\ev{H}$, we have $f\ev{u}f\ev{v} \in E\ev{G}$. 

\item Let $\mathrm{hom}\ev{H,G}$ be the number of graph homomorphisms $H \rightarrow G$. 

\item The \textit{homomorphism density} of $H$ in $G$ is the fraction
$$
t_{H}\ev{G} := \dfrac{\mathrm{hom}\ev{H,G}}{|V\ev{G}|^{|V\ev{H}|}}
$$
\end{enumerate}
\end{definition}

\begin{conjecture}[Sidorenko's Conjecture \mbox{\cite{sidorenko}}]
For every bipartite graph $H$ and for any graph $G$, we have:
$$
t_H\ev{G} \geq t_{K_2}\ev{G}^{|E\ev{H}|}
$$
\end{conjecture}


Since Sidorenko's conjecture is open in general, it is natural to seek a scalable weakening that recovers the full conjecture in the limit. We obtain such a weakening by bounding the sizes of the two sides of the bipartition: for a parameter $k$, we restrict attention to bipartite graphs in which each side has at most $k$ vertices. As $k$ grows, this class exhausts all bipartite graphs, so the family of statements below interpolates between small, tractable cases and the full conjecture.

\begin{openproblem}[Sidorenko's Conjecture for bipartite graphs with bounded sides]
Let $k \in \N$. Prove that Sidorenko's conjecture holds for every bipartite graph $H$ that admits a bipartition $V\ev{H} = X \cup Y$ with $|X| \leq k$ and $|Y| \leq k$; that is, for every such $H$ and any graph $G$, we have
$$
t_H\ev{G} \geq t_{K_2}\ev{G}^{|E\ev{H}|}.
$$
\end{openproblem}
