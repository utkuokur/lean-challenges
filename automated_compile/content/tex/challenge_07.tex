\section*{The Sphere Packing Problem}
\refstepcounter{section}

\begin{definition}[Packing, density, sphere-packing constant]
Fix a dimension $d \in \N$.
\begin{enumerate}
\item A \emph{unit packing} of $\R^d$ is a set $C \subseteq \R^d$ of centers with $\lVert x - y \rVert \geq 2$ for all distinct $x,y \in C$; equivalently the open unit balls $B\ev{x,1}$, $x \in C$, are pairwise disjoint.
\item The \emph{upper density} of $C$ is
$$
\bar\delta\ev{C} := \limsup_{R \to \infty} \frac{\mathrm{vol}\!\ew{\textstyle\bigcup_{x \in C} B\ev{x,1} \cap B\ev{0,R}}}{\mathrm{vol}\!\ev{B\ev{0,R}}}.
$$
\item The \emph{sphere-packing constant} $\Delta_d := \sup_C \bar\delta\ev{C}$ is the supremum over all unit packings $C$; the supremum over packings whose centre set is a lattice is the \emph{lattice} packing constant $\Delta_d^{L}$.
\end{enumerate}
\end{definition}

\begin{openproblem}[The sphere-packing problem]
Given $r\geq 0$, show that $\Delta_r = \Delta_r^{L}$.
\end{openproblem}

The value of $\Delta_r$ is known only for $r \in \{1,2,3,8,24\}$. The case $r = 3$ is the Kepler conjecture, proven by \cite{hales2005kepler}. The cases $r = 8$ and $r = 24$ were settled by Viazovska \cite{viazovska2017eight} and by Cohn, Kumar, Miller, Radchenko and Viazovska \cite{cohn2017twentyfour}. In each solved case the optimum is attained by a lattice ($\Z$, $A_2$, the FCC lattice, $E_8$, and the Leech lattice). The largest dimension solved is $r = 24$.
